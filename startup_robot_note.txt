#find this service
sudo gedit /etc/systemd/system/startup_robot.service


#within the service:

	[Unit]
	Description=Start ATMO using m4 jetson
	After=multi-user.target

	[Service]
	Type=oneshot
	RemainAfterExit=yes
	ExecStart=/bin/bash /home/m4/ATMO/ATMO/atmo_ws/startup_robot.sh
	WorkingDirectory=/home/m4/ATMO/ATMO/atmo_ws
	User=m4
	Environment="PATH=/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin"
	StandardOutput=append:/home/m4/ATMO/ATMO/startup_robot.log
	StandardError=append:/home/m4/ATMO/ATMO/startup_robot.log
	TimeoutStartSec=120

	[Install]
	WantedBy=multi-user.target


#start service
sudo systemctl daemon-reload
sudo systemctl restart start_robot.service

#reload service
sudo systemctl restart start_robot.service

#disable service
sudo systemctl disable start_robot.service
